  - file: [src/cli/daemon-cli/restart-health.test.ts:242](https://github.com/openclaw/openclaw/blob/01bc49c88c0cef9807b15df4d7486a0c5fe4afbd/src/cli/daemon-cli/restart-health.test.ts#L242)
